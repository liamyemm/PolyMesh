214  std::vector<double> regularity()
215  {
222 
223  std::vector<std::vector<double>> reg_cell(n_cells(), {0.0, 0.0});
224  std::size_t count = 0;
225  for (auto &T : _cells)
226  {
227  double hT = T->diam();
228  VectorRd xT = T->center_mass();
229 
230  reg_cell[count][0] = hT / pow(T->measure(), 1.0 / DIMENSION);
231 
232  double rhoT = hT;
233  std::vector<Face *> faces = T->get_faces();
234  for (auto &F : faces)
235  {
236  double hF = F->diam();
237  VectorRd xF = F->center_mass();
238  VectorRd nTF = F->normal(); // sign does not matter
239 
240  reg_cell[count][0] = std::max(reg_cell[count][0], hT / hF);
241 
242  rhoT = std::min(rhoT, std::abs((xT - xF).dot(nTF))); // If xT is not in T, is this really a good measure?
243  }
244  reg_cell[count][1] = hT / rhoT;
245  ++count; // could just use iterators
246  }
247 
248  std::vector<double> value(2, 0.0);
249  for (size_t iT = 0; iT < n_cells(); iT++)
250  {
251  value[0] = std::max(value[0], reg_cell[iT][0]);
252  value[1] = std::max(value[1], reg_cell[iT][1]);
253  }
254 
255  return value;

309  size_t find_cell(const VectorRd x) const
310  {
311 
312  // Locate neighbouring cells
313  std::vector<size_t> neighbouring_cells;
314  for (Cell *T : get_cells())
315  {
316  if ((T->center_mass() - x).norm() <= T->diam())
317  {
318  neighbouring_cells.push_back(T->global_index());
319  }
320  }
321 
322  // In neighbouring cells, find one such that x is contained in one of the subtriangles
323  size_t i = 0;
324  bool found = false;
325  size_t iT = 0;
326  while (i < neighbouring_cells.size() && !found)
327  {
328  iT = neighbouring_cells[i];
329  Cell *T = cell(iT);
330 
331  for(auto& simplex : T->get_simplices())
332  {
333  double area1 = signed_area(x, simplex[0], simplex[1]);
334  double area2 = signed_area(x, simplex[1], simplex[2]);
335  double area3 = signed_area(x, simplex[3], simplex[1]);
336 
337  found = (area1 >= 0.0 && area2 >= 0.0 && area3 >= 0.0) || (area1 <= 0.0 && area2 <= 0.0 && area3 <= 0.0); // if all non neg or non pos must be in or on triangle
338  if(found) break;
339  }
340  ++i;
341  }
342  assert(found);
343 
344  return iT;
