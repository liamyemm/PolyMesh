164  template <size_t object_dim>
165  VectorRd simplex_center_mass(Simplex<object_dim> simplex)
166  {
167  VectorRd center_mass = VectorRd::Zero();
168 
169  for (auto &coord : simplex)
170  {
171  for (size_t i = 0; i < DIMENSION; ++i)
172  {
173  center_mass(i) += coord(i);
174  }
175  }
176  return center_mass / (object_dim + 1);
177  }

179  template <size_t object_dim>
180  double simplex_measure(Simplex<object_dim> simplex)
181  {
182  Eigen::MatrixXd CM_mat = Eigen::MatrixXd::Zero(object_dim + 2, object_dim + 2);
183  for (size_t i = 0; i < object_dim + 1; ++i)
184  {
185  VectorRd i_coords = simplex[i];
186  for (size_t j = i + 1; j < object_dim + 1; ++j)
187  {
188  VectorRd j_coords = simplex[j];
189  double norm = (j_coords - i_coords).norm();
190  CM_mat(i, j) = norm * norm;
191  CM_mat(j, i) = CM_mat(i, j);
192  }
193  CM_mat(i, object_dim + 1) = 1;
194  CM_mat(object_dim + 1, i) = 1;
195  }
196 
197  // Calculate Cayley-Menger determinant
198  double det = CM_mat.determinant();
199 
200  double scaling = std::pow(Math::factorial(object_dim), 2) * std::pow(2, object_dim);
201 
202  return std::sqrt(std::abs(det / scaling));
203  }

205  template <size_t object_dim>
206  Polytope<object_dim>::Polytope(size_t index, Simplices<object_dim> simplices)
207  : _index(index), _simplices(simplices)
208  {
209  assert(object_dim <= DIMENSION);
210 
211  if (object_dim == 0 || object_dim == 1)
212  {
213  assert(simplices.size() == 1);
214  }
215 
216  _is_boundary = false; // by default, is_boundary is set to false. If polytope is on boundary, set is_boundary to true upon construction of the mesh.
217 
218  _measure = 0.0;
219  _center_mass = VectorRd::Zero();
220  std::vector<VectorRd> vertex_coords;
221  for (auto &simplex : simplices)
222  {
223  // assert(simplex.size() == _dim + 1);
224  double measure_of_simplex = simplex_measure<object_dim>(simplex);
225  _measure += measure_of_simplex;
226  _center_mass += measure_of_simplex * simplex_center_mass<object_dim>(simplex);
227  for (auto &coord : simplex)
228  {
229  if (std::find(vertex_coords.begin(), vertex_coords.end(), coord) == vertex_coords.end())
230  {
231  vertex_coords.push_back(coord); // if coord not already in vertex_coords, add it
232  }
233  }
234  }
235  _center_mass /= _measure;
236 
237  _diameter = 0.0;
238  for (auto it = vertex_coords.begin(); it != vertex_coords.end(); ++it)
239  {
240  for (auto jt = it; jt != vertex_coords.end(); ++jt)
241  {
242  _diameter = std::max(_diameter, (*it - *jt).norm());
243  }
244  }
245 
246  if (object_dim == DIMENSION - 1) // find the tangent and normal
247  {
248  _tangent = (this->get_simplices()[0][1] - this->get_simplices()[0][0]).normalized();
249  _normal = VectorRd(-_tangent(1), _tangent(0));
250  }
251  }

328  template <size_t object_dim>
329  void Polytope<object_dim>::construct_face_normals() // not very efficient - probably room for improvement
330  {
331  assert(object_dim == DIMENSION);
332  for (size_t iF = 0; iF < _edges.size(); ++iF)
333  {
334  VectorRd normal = _edges[iF]->normal();
335  Simplex<object_dim - 1> face_simplex = _edges[iF]->get_simplices()[0];
336  VectorRd center = VectorRd::Zero();
337  double count;
338  for (auto &cell_simplex : this->get_simplices())
339  {
340  count = 0;
341  for (size_t i = 0; (i < cell_simplex.size()) && count < 2; ++i)
342  {
343  if (std::find(face_simplex.begin(), face_simplex.end(), cell_simplex[i]) == face_simplex.end()) // requires numerical precision
344  {
345  ++count;
346  }
347  }
348  if (count == 1) // only don't share one coordinate
349  {
350  center = simplex_center_mass<object_dim>(cell_simplex);
351  break;
352  }
353  }
354  assert(count == 1);
355  // _face_directions.push_back(Math::sgn((_faces[iF]->center_mass() - _center_mass).dot(normal))); // star shaped wrt center mass
356  _face_directions.push_back(Math::sgn((_edges[iF]->center_mass() - center).dot(normal)));
357  assert(_face_directions[iF] != 0);
358  }
359  }
