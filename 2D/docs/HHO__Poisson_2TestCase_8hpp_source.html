72  inline void reorder_edges(Mesh *mesh_ptr)
73  {
74  // Create vector with all non-Dirichlet edges first, and all Dirichlet edges at the end
75  std::vector<size_t> new_to_old(mesh_ptr->n_edges(), 0);
76  // Index for non-Dirichlet edges start at 0 and increases, index for Dirichlet edges start at n_edges()-1
77  // and decreases
78  size_t idx_nondir = 0;
79  size_t idx_dir = mesh_ptr->n_edges() - 1;
80  for (auto &edge : mesh_ptr->get_edges())
81  {
82  assert(idx_nondir < idx_dir + 1);
83  if (edge->is_boundary())
84  {
85  new_to_old[idx_dir] = edge->global_index();
86  idx_dir--;
87  }
88  else
89  {
90  new_to_old[idx_nondir] = edge->global_index();
91  idx_nondir++;
92  }
93  }
94 
95  // Check: idx_dir and idx_nondir must just have crossed
96  assert(idx_nondir == idx_dir + 1);
97 
98  // Reordering
99  mesh_ptr->renum('E', new_to_old);
100  }
101 
103  }
