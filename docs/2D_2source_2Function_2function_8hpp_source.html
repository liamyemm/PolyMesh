43  template <typename PointType>
44  std::vector<Pole<PointType>> combine_poles(const std::vector<Pole<PointType>> &poles1, const std::vector<Pole<PointType>> &poles2)
45  {
46  std::vector<Pole<PointType>> combinedPoles;
47 
48  // Copy poles from the first vector
49  combinedPoles.insert(combinedPoles.end(), poles1.begin(), poles1.end());
50 
51  // Check poles from the second vector
52  for (const auto &pole2 : poles2)
53  {
54  bool found = false;
55 
56  // Search for a pole with the same location
57  for (auto &pole1 : combinedPoles)
58  {
59  PointType pole_diff(pole1.location - pole2.location);
60  if (Math::norm(pole_diff) < 1E-12)
61  {
62  // Pole with the same location found, combine the orders
63  pole1.order = pole1.order + pole2.order;
64  found = true;
65  break;
66  }
67  }
68 
69  // If no pole with the same location was found, add it to the combined poles
70  if (!found)
71  {
72  combinedPoles.push_back(pole2);
73  }
74  }
75 
76  return combinedPoles;
77  }
