67  {
68  // Create vector with all non-Dirichlet edges first, and all Dirichlet edges at the end
69  std::vector<size_t> new_to_old(mesh_ptr->n_edges(), 0);
70  // Index for non-Dirichlet edges start at 0 and increases, index for Dirichlet edges start at n_edges()-1
71  // and decreases
72  size_t idx_nondir = 0;
73  size_t idx_dir = mesh_ptr->n_edges() - 1;
74  for (auto &edge : mesh_ptr->get_edges())
75  {
76  assert(idx_nondir < idx_dir + 1);
77  if (edge->is_boundary())
78  {
79  new_to_old[idx_dir] = edge->global_index();
80  idx_dir--;
81  }
