62  inline void reorder_edges(Mesh *mesh_ptr)
63  {
64  // Create vector with all non-Dirichlet edges first, and all Dirichlet edges at the end
65  std::vector<size_t> new_to_old(mesh_ptr->n_edges(), 0);
66  // Index for non-Dirichlet edges start at 0 and increases, index for Dirichlet edges start at n_edges()-1
67  // and decreases
68  size_t idx_nondir = 0;
69  size_t idx_dir = mesh_ptr->n_edges() - 1;
70  for (auto &edge : mesh_ptr->get_edges())
71  {
72  assert(idx_nondir < idx_dir + 1);
73  if (edge->is_boundary())
74  {
75  new_to_old[idx_dir] = edge->global_index();
76  idx_dir--;
77  }
78  else
79  {
80  new_to_old[idx_nondir] = edge->global_index();
81  idx_nondir++;
82  }
83  }
84 
85  // Check: idx_dir and idx_nondir must just have crossed
86  assert(idx_nondir == idx_dir + 1);
87 
88  // Reordering
89  mesh_ptr->renum('E', new_to_old);
90  }
91 
93  }
94 }
